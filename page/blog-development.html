Cache de Sprites:

class SpriteCache:
    _cache = {}
    
    @classmethod
    def get_sprite(cls, path, size):
        key = (path, size)
        if key not in cls._cache:
            sprite = pygame.image.load(path)
            sprite = pygame.transform.scale(sprite, size)
            cls._cache[key] = sprite
        return cls._cache[key]
